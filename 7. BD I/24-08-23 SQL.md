## Atividade prática

```postgresql
-- biblioteca
DROP TABLE livros;

CREATE TABLE livros (
	livro_id SERIAL PRIMARY KEY, 
	titulo VARCHAR(200) NOT NULL, 
	autor VARCHAR(150) NOT NULL, 
	ano_publicacao INT,
	quantidade_disponivel INT DEFAULT 1
);

INSERT INTO livros (titulo, autor)
VALUES ('O Alquimista', 'Paulo Coelho');

SELECT * FROM livros;
```
